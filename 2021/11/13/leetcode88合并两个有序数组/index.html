leetcode88合并两个有序数组

 2021-11-13 16:23:39

leetcode 88.给你两个按 非递减顺序 排列的整数数组 nums1 和 nums2，另有两个整数 m 和 n ，分别表示 nums1 和 nums2 中的元素数目。

请你 合并 nums2 到 nums1 中，使合并后的数组同样按 非递减顺序 排列。
注意：最终，合并后数组不应由函数返回，而是存储在数组 nums1 中。为了应对这种情况，nums1 的初始长度为 m + n，其中前 m 个元素表示应合并的元素，后 n 个元素为 0 ，应忽略。nums2 的长度为 n 。

package com.ruoan.study.leetcode;

import java.util.stream.Stream;

public class Solution88 {
    public static void main(String[] args) {
        case1();
        case2();
        case3();
    }

    private static void case1() {
        int[] nums1 = Stream.of(1, 2, 3, 0, 0, 0).mapToInt(Integer::intValue).toArray();
        int[] nums2 = Stream.of(2, 5, 6).mapToInt(Integer::intValue).toArray();
        Solution88 solution = new Solution88();
        solution.merge(nums1, 3, nums2, 3);
        print(nums1);
    }

    private static void case2() {
        int[] nums1 = Stream.of(0).mapToInt(Integer::intValue).toArray();
        int[] nums2 = Stream.of(1).mapToInt(Integer::intValue).toArray();
        Solution88 solution = new Solution88();
        solution.merge2(nums1, 0, nums2, 1);
        print(nums1);
    }

    private static void case3() {
        int[] nums1 = Stream.of(2, 0).mapToInt(Integer::intValue).toArray();
        int[] nums2 = Stream.of(1).mapToInt(Integer::intValue).toArray();
        Solution88 solution88 = new Solution88();
        solution88.merge(nums1, 1, nums2, 1);
        print(nums1);
    }

    private static void print(int[] nums1) {
        for (int i : nums1) {
            System.out.println(i);
        }
        System.out.println("------------");
    }

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        for (int i = nums1.length - 1; i >= 0; i--) {
            if (n <= 0) {
                break;
            }
            if (m == 0) {
                for (int j = 0; j < n; j++) {
                    nums1[j] = nums2[j];
                }
                return;
            }
            if (nums1[m - 1] <= nums2[n - 1]) {
                nums1[i] = nums2[n - 1];
                n--;
            } else {
                nums1[i] = nums1[m - 1];
                nums1[m - 1] = 0;
                m--;
            }
        }
    }

    public void merge2(int[] nums1, int m, int[] nums2, int n) {
        int p1 = m - 1;
        int p2 = n - 1;
        for (int i = nums1.length - 1; i >= 0; i--) {
            if (p1 < 0) {
                nums1[i] = nums2[p2];
                p2--;
            } else if (p2 < 0) {
                nums1[i] = nums1[p1];
                p1--;
            } else if (nums1[p1] < nums2[p2]) {
                nums1[i] = nums2[p2];
                p2--;
            } else {
                nums1[i] = nums1[p1];
                p1--;
            }
        }
    }
}
