运用你所掌握的数据结构，设计和实现一个  LRU (最近最少使用) 缓存机制 。
实现 LRUCache 类：

LRUCache(int capacity) 以正整数作为容量 capacity 初始化 LRU 缓存
int get(int key) 如果关键字 key 存在于缓存中，则返回关键字的值，否则返回 -1 。
void put(int key, int value) 如果关键字已经存在，则变更其数据值；如果关键字不存在，则插入该组「关键字-值」。当缓存容量达到上限时，它应该在写入新数据之前删除最久未使用的数据值，从而为新的数据值留出空间。  

进阶：你是否可以在 O(1) 时间复杂度内完成这两种操作？

package com.ruoan.study.leetcode;

import java.util.HashMap;
import java.util.Map;

class Solution146 {

    private Map<Integer, Node> cache;
    private DoubleLinkedList doubleLinkedList;
    private int capacity;

    public Solution146(int capacity) {
        doubleLinkedList = new DoubleLinkedList();
        cache = new HashMap();
        this.capacity = capacity;
    }

    public int get(int key) {
        if (!cache.containsKey(key)) {
            return -1;
        }
        Node target = cache.get(key);
        put(key, target.value);
        return target.value;
    }

    public void put(int key, int value) {
        //缓存当中已经有了这个节点，则对其进行更新
        if (cache.containsKey(key)) {
            //1.先把旧节点拿到
            Node target = cache.get(key);
            //2. 把这个旧节点从双端链表删除
            doubleLinkedList.delete(target);
            //3.构造新节点
            Node newNode = new Node(key, value);
            cache.put(key, newNode);
            //4. 把这个节点放在双端链表的头部
            doubleLinkedList.addToHead(newNode);
        } else {
            if (cache.size() == capacity) {
                int k = doubleLinkedList.deleteFromTail();
                cache.remove(k);
            }
            //todo 相同代码，可以合并
            Node newNode = new Node(key, value);
            //1.把新节点放在HashMap缓存当中
            cache.put(key, newNode);
            //2.把新节点添加到双端链表中
            doubleLinkedList.addToHead(newNode);
        }
    }

    public static void main(String[] args) {
        Solution146 lRUCache = new Solution146(2);
        lRUCache.put(1, 1); // 缓存是 {1=1}
        lRUCache.put(2, 2); // 缓存是 {1=1, 2=2}
        System.out.println(lRUCache.get(1));
        lRUCache.put(3, 3); // 该操作会使得关键字 2 作废，缓存是 {1=1, 3=3}
        System.out.println(lRUCache.get(2));
        lRUCache.put(4, 4); // 该操作会使得关键字 1 作废，缓存是 {4=4, 3=3}
        lRUCache.get(1);    // 返回 -1 (未找到)
        lRUCache.get(3);    // 返回 3
        lRUCache.get(4);    // 返回 4
    }
}

//1. 存储结构: HashMap<Integer,Node>
//2. 构造双向链表, 双向链表需要提供插入和删除，插入节点一律插在头部，删除节点一律从尾部删除

class Node {
    int key;
    int value;
    Node next;
    Node pre;

    Node(int key, int value) {
        this.key = key;
        this.value = value;
    }
}

class DoubleLinkedList {
    Node head;
    Node tail;

    DoubleLinkedList() {
        head = new Node(0, 0);
        tail = new Node(0, 0);
        head.next = tail;
        tail.pre = head;
    }

    void addToHead(Node newNode) {
        newNode.next = head.next;
        head.next.pre = newNode;
        head.next = newNode;
        newNode.pre = head;
    }

    int deleteFromTail() {
        if (tail.pre == head) return -1;
        return delete(tail.pre);
    }

    int delete(Node node) {
        node.next.pre = node.pre;
        node.pre.next = node.next;
        return node.key;
    }
}

 * Your LRUCache object will be instantiated and called as such:
 * LRUCache obj = new LRUCache(capacity);
 * int param_1 = obj.get(key);
 * obj.put(key,value);
